        function toggleAttributes() {
            var i, len,
                elem,
                elems = document.getElementsByClassName("toggle");

            for (i = 0, len = elems.length; i < len; i += 1) {
                elem = elems[i];
                if (elem.style.visibility === "hidden") {
                    elem.style.visibility = "";
                } else {
                    elem.style.visibility = "hidden";
                }
            }
